class Solution {
    public boolean Check(int num){
        int range = (int)Math.sqrt(num);
        for(int i = 2; i <= range; i++){
            if(num % i == 0) return false;
        }
        return true;
    }
    public int solution(int n) {
        int answer = 0;
        for(int i = 2; i <= n; i++){
            if(Check(i)) answer++;
        }
        return answer;
    }
}
