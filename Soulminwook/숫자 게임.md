import java.util.Arrays;

class Solution {
    public int solution(int[] A, int[] B) {
        int size = A.length;
        Arrays.sort(A);
        Arrays.sort(B);
        int answer = 0, index_A = 0, index_B = 0;
        while(index_A < size && index_B < size){
            if(A[index_A] < B[index_B]){
                answer++;
                index_A++;
                index_B++;
            } else {
                index_B++;
            }
        }
        return answer;
    }
}
