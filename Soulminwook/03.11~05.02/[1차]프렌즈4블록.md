import java.util.*;

class Solution {
    public int answer;
    public char[][] puzzle;
    public int Check(char[][] puzzle, int m, int n){
        int result = 0;
        boolean[][] visited = new boolean[m][n];
        for(int i = 0; i < m - 1; i++){
            for(int j = 0; j < n - 1; j++){
                if(puzzle[i][j] == '#') continue;
                if(puzzle[i][j] == puzzle[i][j + 1] && puzzle[i][j] == puzzle[i + 1][j] && puzzle[i][j] == puzzle[i + 1][j + 1]){
                    visited[i][j] = true;
                    visited[i][j + 1] = true;
                    visited[i + 1][j] = true;
                    visited[i + 1][j + 1] = true;
                }
            }
        }
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                if(visited[i][j]){
                    result++;
                    puzzle[i][j] = '#';
                }
            }
        }
        answer += result;
        return result;
    }
    public void Change(char[][] puzzle, int m, int n){
        for(int j = 0; j < n; j++){
            int start = -1;
            for(int i = m - 1; i >= 0; i--){
                if(puzzle[i][j] == '#'){
                    start = i;
                    break;
                }
            }
            for(int i = start; i >= 0; i--){
                if(puzzle[i][j] != '#'){
                    char tmp = puzzle[i][j];
                    puzzle[i][j] = puzzle[start][j];
                    puzzle[start--][j] = tmp;
                }
            }
        }
    }
    public int solution(int m, int n, String[] board) {
        answer = 0;
        puzzle = new char[m + 1][n + 1];
        int x_cnt = 0, y_cnt = 0;
        for(String str : board){
            y_cnt = 0;
            for(char word : str.toCharArray()){
                puzzle[x_cnt][y_cnt++] = word;
            }
            x_cnt++;
        }
        while(Check(puzzle, m + 1, n + 1) > 0){
            Change(puzzle, m + 1, n + 1);
        }
        return answer;
    }
}
