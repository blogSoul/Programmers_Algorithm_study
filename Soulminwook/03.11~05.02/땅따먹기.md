import java.util.*;

class Solution {
    public int solution(int[][] land) {
        int land_size = land.length, answer = 0;
        int[] dp = new int[4];
        for(int i = 1; i < land_size; i++){
            Arrays.fill(dp, 0);
            for(int j = 0; j < 4; j++){
                for(int k = 0; k < 4; k++){
                    if(j == k) continue;
                    dp[j] = Math.max(dp[j], land[i - 1][k]);
                }
            }    
            for(int j = 0; j < 4; j++){
                land[i][j] += dp[j];   
            }
        }
        for(int j = 0; j < 4; j++){
            answer = Math.max(answer, land[land_size - 1][j]);
        }
        return answer;
    }
}
