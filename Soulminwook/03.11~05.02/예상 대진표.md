class Solution {
    public int solution(int n, int a, int b) {
        int answer = 1, first = a, second = b;
        if(a > b){
            first = b;
            second = a;
        }
        while(first < second){
            first = (first % 2 == 1) ? (first + 1) / 2 : first / 2;
            second = (second % 2 == 1) ? (second + 1) / 2 : second / 2;
            answer++;
        }
        return answer - 1;
    }
}
