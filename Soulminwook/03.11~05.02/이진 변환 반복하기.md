class Solution {
    public int[] solution(String s) {
        int[] answer = {0, 0};
        while(!s.equals("1")){
            int count = 0;
            for(char i : s.toCharArray()){
                if(i == '1'){
                    count++;
                } else if(i == '0'){
                    answer[1]++;
                }
            }
            s = Integer.toBinaryString(count);    
            answer[0]++;
        }
        return answer;
    }
}
