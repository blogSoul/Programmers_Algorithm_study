import java.util.*;

class Solution {
    public ArrayList<ArrayList<Integer>> graph;
    public void initGraph(int n){
        graph = new ArrayList<>();
        for(int i = 0; i < n; i++){
            graph.add(new ArrayList<Integer>());
        }
    }
    public boolean solution(int n, int[][] path, int[][] order) {
        initGraph(n);
        for(int[] edge : path){
            graph.get(edge[0]).add(edge[1]);
            graph.get(edge[1]).add(edge[0]);
        }
        int[] before = new int[n];
        for(int[] edge : order){
            before[edge[1]] = edge[0];
        }
        if(before[0] != 0) return false;
        Queue<Integer> queue = new LinkedList<>();
        for(int next : graph.get(0)){
            queue.offer(next);
        }
        queue.offer(0);
        int[] save = new int[n];
        boolean[] visited = new boolean[n];
        visited[0] = true;
        while(!queue.isEmpty()){
            int node = queue.poll();
            if(visited[node]) continue;
            if(!visited[before[node]]){
                save[before[node]] = node;
                continue;
            }
            visited[node] = true;
            for(int next : graph.get(node)){
                queue.offer(next);
            }
            queue.offer(save[node]);
        }
        for(boolean isVisited : visited){
            if(!isVisited) return false;
        }
        return true;
    }
}

class Solution {
    public boolean flag;
    public boolean[] visited, finished;
    public ArrayList<ArrayList<Integer>> graph, checkgraph;
    public void initGraph(int n){
        graph = new ArrayList<>();

            graph.add(new ArrayList<Integer>());
            checkgraph.add(new ArrayList<Integer>());
        }
    }
    public void dfs(int index){
        if(!flag) return;
        visited[index] = true;
        for(int next : checkgraph.get(index)){
            if(!visited[next]){
                dfs(next);
            } else if(!finished[next]){
                flag = false;
            }
        }
        finished[index] = true;
    }

        }
