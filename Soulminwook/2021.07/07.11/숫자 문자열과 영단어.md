class Solution {
    public int solution(String s) {
        String[] number = {"zero", "one", "two", "three", "four", "five", "six", "seven", "eight", "nine"};
        for(int i = 0; i < 10;){
            int num = s.indexOf(number[i]);
            if(num != -1){
                s = s.substring(0, num) + i + s.substring(num + number[i].length());
            } else {
                i++;
            }
        }
        return Integer.parseInt(s);
    }
}
