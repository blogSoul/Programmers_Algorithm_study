class Solution {
    public int stoi(String str){
        return Integer.parseInt(str);
    }
    public int musicsize(String start, String end){
        String[] start_info = start.split(":");
        String[] end_info = end.split(":");
        int starttime = stoi(start_info[0]) * 60 + stoi(start_info[1]);
        int endtime = stoi(end_info[0]) * 60 + stoi(end_info[1]);
        return endtime - starttime;
    }
    public String solution(String m, String[] musicinfos) {
        int m_size = m.length();
        String answer = "";
        for(String musicinfo : musicinfos){
            String[] music_info = musicinfo.split(",");
            int music_size = musicsize(music_info[0], music_info[1]), titlesize = music_info[3].length();
            String str = "";
            int cycle = music_size / titlesize, remain = music_size % titlesize;
            for(int i = 0; i < cycle; i++){
                str += music_info[3];
            }
            str += music_info[3].substring(0, remain);
            if(str.contains(m)){
                int index = str.indexOf(m) + m_size;
                if(index >= str.length()){
                    continue;
                }
                if(str.charAt(index) == '#'){
                    continue;
                }
                if(music_info[2].length() > answer.length()){
                    answer = music_info[2];
                }
                System.out.println(str + " " + str.indexOf(index) + " " + answer);
            }
        }
        if(answer.equals("")){
            answer = "(None)";
        }
        return answer;
    }
}
