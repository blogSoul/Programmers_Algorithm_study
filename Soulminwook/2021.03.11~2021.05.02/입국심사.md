import java.util.*;

class Solution {
    public long BinarySearch(long start, long end,int[] times,long n){
        while(start < end){
            long mid = (start + end) / 2, sum = 0;
            for(int time : times){
                if(time > mid) break;
                sum += mid / time;
            }
            if(n <= sum){
                end = mid;
            } else {
                start = mid + 1;
            }
        }
        return start;
    }
    public long solution(int n, int[] times) {
        Arrays.sort(times);
        return BinarySearch(0, (long)n * (long)times[times.length - 1], times, n);
    }
}
