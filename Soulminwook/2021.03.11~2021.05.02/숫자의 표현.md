class Solution {
    public int solution(int n) {
        int answer = 0;
        for(int start = 1; start <= n; start++){
            for(int count = 1; count <= n; count++){
                int sum = (start * count) + ((count - 1) * count ) / 2;
                if(sum < n){
                    continue;
                } else if(sum == n){
                    answer++;
                    continue;
                } else {
                    break;
                }
            }
        }
        return answer;
    }
}
