import java.util.*;

class Solution {
    public int[] result;
    public void Check(ArrayList<Integer> arr, int[][] results, int n) {
        for(int[] element : results){
            int a = arr.indexOf(element[0]), b = arr.indexOf(element[1]);
            if(a < b){
                return;
            }
        }
        for(int i = 0; i < n; i++){
            if(result[i] == -1){
                continue;
            } else if(result[i] == 0){
                result[i] = arr.get(i);
            } else if(result[i] > 0 & result[i] != arr.get(i)) {
                result[i] = -1;
            }
        }
    }
    public void permutation(int count, int n, ArrayList<Integer> arr, int[][] results, boolean[] visited) {
        if(count == n){
            Check(arr, results, n);
            return;
        }
        for(int i = 1; i <= n; i++){
            if(!visited[i]){
                visited[i] = true;
                arr.add(i);
                permutation(count + 1, n, arr, results, visited);
                arr.remove(arr.size() - 1);
                visited[i] = false;
            }
        }
    }
    public int solution(int n, int[][] results) {
        int answer = 0;
        ArrayList<Integer> arr = new ArrayList<>();
        result = new int[n];
        boolean[] visited = new boolean[n + 1];
        permutation(0, n, arr, results, visited);
        for(int i = 0; i < n; i++){
            if(result[i] == -1){
                continue;
            }
            answer++;
        }
        return answer;
    }
}
