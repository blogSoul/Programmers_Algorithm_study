```java
class Solution {
    public int[] solution(int n, long k) {
        int[] answer = new int[n];
        long[] count = new long[n];
        int num = 1;
        for(int i = 0; i < n; i++){
            count[n - 1 - i] = num;
            num *= (i < 1) ? 1 : num + 1;
        }
        for(long i : count){
            System.out.print(i + " ");
        }
        for(int i = 0; i < n; i++){
            long number = k / count[i] + 1;
            answer[i] = (int)number;
            k -= (k / count[i]) * count[i];
        }
        return answer;
    }
}
```

시간 초과 ... 할 수 있는 다른 방법을 생각해보자

```java
class Solution {
    public long answer_cnt;
    public int[] result;
    public void permutation(int count, int N, long k, int[] answer, boolean[] visited){
        if(result[0] != 0){
            return;
        }
        if(count == N){
            answer_cnt++;
            if(k == answer_cnt){
                int num = 0;
                for(int i : answer){
                    result[num++] = i;
                }
            }
            return;
        }
        for(int i = 0; i < N; i++){
            if(!visited[i]){
                visited[i] = true;
                answer[count] = i + 1;
                permutation(count + 1, N, k, answer, visited);
                answer[count] = 0;
                visited[i] = false;
            }
        }
    }
    public int[] solution(int n, long k) {
        answer_cnt = 0;
        result = new int[n];
        int[] answer = new int[n];
        boolean[] visited = new boolean[n];
        permutation(0, n, k, answer, visited);
        return result;
    }
}
```
