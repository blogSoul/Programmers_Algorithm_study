class Solution {
    public int[] solution(int[] arr) {
        int size = arr.length;
        if(size == 1){
            int[] answer = { -1 };
            return answer;    
        }
        int min = Integer.MAX_VALUE, cnt = 0;
        for(int i = 0; i < size; i++){
            if(min > arr[i]) min = arr[i];
        }
        int[] answer = new int[size - 1];
        for(int i = 0; i < size; i++){
            if(min < arr[i]) answer[cnt++] = arr[i];
        }
        return answer;
    }
}
