class Solution {
    public int Count(long num){
        if(num == 1) return 0;
        for(long i = 2; i < num; i++){
            if(num % i == 0){
                return (int)(num / i);
            }
        }
        return 1;
    }
    public int[] solution(long begin, long end) {
        int range = (int)(end - begin) + 1, cnt = 0;
        int[] answer = new int[range];
        for(long i = begin; i <= end; i++){
            answer[cnt++] = Count(i);
        }
        return answer;
    }
}
