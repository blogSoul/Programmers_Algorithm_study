import java.util.Arrays;

class Solution {
    public long solution(long n) {
        char[] list = Long.toString(n).toCharArray();
        Arrays.sort(list);
        String str = "";
        for(char word : list){
            str = word + str;
        }
        return Long.parseLong(str);
    }
}
