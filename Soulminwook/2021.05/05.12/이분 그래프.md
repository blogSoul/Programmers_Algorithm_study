import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.*;

class Main {
    public static boolean CheckByBfs(ArrayList<ArrayList<Integer>> graph, HashSet<Integer> set, int V){
        int start = 0;
        int[] visited = new int[V + 1];
        for(int num : set){
            if(visited[num] == 0){
                start = num;
                break;
            }
        }
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(start);
        visited[start] = 1;
        set.remove(start);
        while(!queue.isEmpty()) {
            int node = queue.poll();
            int groupNumber = visited[node];
            for (int nextNode : graph.get(node)) {
                if (visited[nextNode] == 0) {
                    if (groupNumber == 1) {
                        visited[nextNode] = 2;
                    } else if (groupNumber == 2) {
                        visited[nextNode] = 1;
                    }
                    set.remove(nextNode);
                    queue.offer(nextNode);
                } else if (visited[nextNode] == 1 && groupNumber == 1) {
                    return false;
                } else if (visited[nextNode] == 2 && groupNumber == 2) {
                    return false;
                }
            }
            if(queue.isEmpty()){
                start = -1;
                for(int num : set){
                    if(visited[num] == 0){
                        start = num;
                        break;
                    }
                }
                if(start == -1){
                    break;
                } else {
                    queue.offer(start);
                    visited[start] = 1;
                }
            }
        }
        return true;
    }
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        int test_cases = Integer.parseInt(br.readLine());
        for(int test_case = 0; test_case < test_cases; test_case++){
            StringTokenizer st = new StringTokenizer(br.readLine(), " ");
            int V = Integer.parseInt(st.nextToken());
            int E = Integer.parseInt(st.nextToken());
            ArrayList<ArrayList<Integer>> graph = new ArrayList<>();
            for(int i = 0; i <= V; i++) {
                graph.add(new ArrayList<>());
            }
            HashSet<Integer> set = new HashSet<>();
            for(int i = 0; i < E; i++) {
                st = new StringTokenizer(br.readLine(), " ");
                int start = Integer.parseInt(st.nextToken());
                int end = Integer.parseInt(st.nextToken());
                graph.get(start).add(end);
                graph.get(end).add(start);
                set.add(start);
                set.add(end);
            }
            if(CheckByBfs(graph, set, V)){
                bw.write("YES\n");
            } else {
                bw.write("NO\n");
            }
        }
        br.close();
        bw.close();
    }
}
