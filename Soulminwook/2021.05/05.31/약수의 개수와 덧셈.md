class Solution {
    public boolean isCountOdd(int num){
        int sqrt = (int)Math.sqrt(num);
        return num == sqrt * sqrt;
    }
    public int solution(int left, int right) {
        int answer = 0;
        for(int i = left; i <= right; i++){
            if(isCountOdd(i)){
                answer -= i;
            } else {
                answer += i;
            }
        }
        return answer;
    }
}
