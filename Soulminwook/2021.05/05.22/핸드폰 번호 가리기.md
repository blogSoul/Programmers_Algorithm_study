class Solution {
    public String solution(String phone_number) {
        String answer = "";
        int phone_number_size = phone_number.length();
        for(int i = 0; i < phone_number_size; i++){
            if(i < phone_number_size - 4){
                answer += "*";
            } else {
                answer += phone_number.charAt(i);
            }
        }
        return answer;
    }
}
