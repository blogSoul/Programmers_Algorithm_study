class Solution {

            if(stack.isEmpty() || word == '('){
                stack.push(word);

                stack.pop();

                return false;

                second++;

            permutation(count + 1, n, first, second, arr);

                first--;
            } else {
                second--;
            }
        }
    }
    public int solution(int n) {
        answer = 0;
        char[] arr = new char[2 * n];
        permutation(0, 2 * n, 0, 0, arr);
