class Solution {
    public int answer, size;
    public void dfs(int[] numbers, int target, int sum, int count){
        if(count == size){
            if(target == sum){
                answer++;
            }
            return;
        }
        dfs(numbers, target, sum - numbers[count], count + 1);
        dfs(numbers, target, sum + numbers[count], count + 1);
    }
    public int solution(int[] numbers, int target) {
        answer = 0;
        size = numbers.length;
        dfs(numbers, target, 0, 0);
        return answer;
    }
}
