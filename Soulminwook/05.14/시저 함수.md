class Solution {
    public String solution(String s, int n) {
        int index = 0;
        String answer = "";
        for(char word : s.toCharArray()){
            if('A' <= word && word <= 'Z'){
                word = (char)('A' + (word - 'A' + n) % 26);
            } else if('a' <= word && word <= 'z'){
                word = (char)('a' + (word - 'a' + n) % 26);
            }
            answer += word;
        }
        return answer;
    }
}
