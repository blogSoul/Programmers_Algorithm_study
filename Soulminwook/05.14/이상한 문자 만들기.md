class Solution {
    public String solution(String s) {
        String answer = "";
        int s_size = s.length(), interval = 'a' - 'A', i = 0;
        for(int k = 0; k < s_size; k++){
            char word = s.charAt(k);
            if(word == ' '){
                i = -1;
            }
            if('A' <= word && word <= 'Z' && i % 2 == 1){
                answer += (char)(word + interval);
            } else if('a' <= word && word <= 'z' && i % 2 == 0){
                answer += (char)(word - interval);
            } else {
                answer += word;
            }
            i++;
        }
        return answer;
    }
}
