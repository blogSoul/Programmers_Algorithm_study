import java.util.ArrayList;
import java.util.Collections;

class Solution {
    public String solution(String s) {
        ArrayList<Character> list = new ArrayList<>();
        for(char word : s.toCharArray()){
            list.add(word);
        }
        Collections.sort(list, Collections.reverseOrder());
        String answer = "";
        for(char word : list){
            answer += word;
        }
        return answer;
    }
}
