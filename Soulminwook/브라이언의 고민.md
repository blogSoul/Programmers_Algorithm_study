import java.util.*;

class Solution {
    public char[] list;
    public String answer;
    public ArrayList<Character> arr;
    public boolean CheckOne(int start, int end, int checkIndex){
        if(checkIndex >= start){
            return false;
        }
        boolean isComplete = true;
        String substr = " ";
        for(int j = start + 1; j < end; j++){
            if(list[j] >= 'a' && list[j] <= 'z'){
                isComplete = false;
                break;
            }
            substr += list[j];
        }
        if(isComplete){
            answer += substr;
            return true;
        }
        return false;
    }
    public boolean CheckTwo(int start, int end, int checkIndex){
        if(checkIndex + 1 >= start){
            return false;
        }
        String substr = " ";
        if(start == 0 || end == list.length - 1){
            return false;
        }
        char word = list[start];
        boolean isBig = true;
        for(int j = start - 1; j <= end + 1; j++){
            if(isBig && list[j] >= 'a' && list[j] <= 'z'){
                if(list[j] != word){
                    return false;
                }
                isBig = false;
                continue;
            }
            substr += list[j];
            isBig = true;
        }
        answer += substr;
        return true;
    }

    public boolean CheckDuplicate(char word){
        if(arr.contains(word)){
            return false;
        }
        arr.add(word);
        return true;
    }
    public String solution(String sentence) {
        list = sentence.toCharArray();
        arr = new ArrayList<Character>();
        int i = 0, size = list.length, start = 0, end = 0, checkIndex = -1;
        answer = "";
        char before = '0';
        for(i = 0; i < size; i++){
            char word = list[i];
            if(word >= 'a' && word <= 'z'){
                if(before == '0'){
                    before = word;
                    start = i;
                } else if(before == word){
                    end = i;
                } else if(before != word){
                    if(!CheckDuplicate(before)){
                        return "invalid";
                    }
                    if(CheckTwo(start, end, checkIndex)){
                        checkIndex = end + 1;
                    } else if(CheckOne(start, end, checkIndex)){
                        checkIndex = end;
                    } else {
                        return "invalid";
                    }
                    start = i;
                    end = i;
                    before = word;
                }
            }
        }
        if(!CheckDuplicate(before)){
            return "invalid";
        }
        if(CheckTwo(start, end, checkIndex)){
            checkIndex = end + 1;
        } else if(CheckOne(start, end, checkIndex)){
            checkIndex = end;
        } else {
            return "invalid";
        }
        if(answer == ""){
            System.out.println("Answer");
            return "invalid";
        }
        return answer.substring(1);
    }
}
