import java.util.*;

class Solution {
    public String Notation(int num, int n){
        String answer = "";
        if(num == 0){
            answer += "0";
            return answer;
        }
        while(num > 0){
            if(num % n == 10){
                answer = "A" + answer;
            } else if(num % n == 11){
                answer = "B" + answer;
            } else if(num % n == 12){
                answer = "C" + answer;
            } else if(num % n == 13){
                answer = "D" + answer;
            } else if(num % n == 14){
                answer = "E" + answer;
            } else if(num % n == 15){
                answer = "F" + answer;
            } else {
                answer = num % n + answer;
            }
            num = num / n;
        }
        return answer;
    }
    public String solution(int n, int t, int m, int p) {
        int cnt = 0;
        String answer = "", result = "";
        while(result.length() <= m * t){
            result += Notation(cnt++, n);
        }
        cnt = 0;
        while(answer.length() < t){
            if(++cnt % m == p % m){
                answer += result.charAt(cnt - 1);
            }
        }
        return answer;
    }
}
