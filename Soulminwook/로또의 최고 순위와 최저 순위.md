class Solution {
    public int[] answer;
    public void Check(int n, int index){
        int num = 0;
        if(n == 6) num = 1;
        else if(n == 5) num = 2;
        else if(n == 4) num = 3;
        else if(n == 3) num = 4;
        else if(n == 2) num = 5;
        else num = 6;
        answer[index] = num;
        return;
    }
    public int[] solution(int[] lottos, int[] win_nums) {
        int count_same = 0, count_zero = 0;
        for(int i = 0; i < 6; i++){
            if(lottos[i] == 0){
                count_zero++;
                continue;
            }
            for(int j = 0; j < 6; j++){
                if(lottos[i] == win_nums[j]){
                    count_same++;
                    break;
                }
            }
        }
        answer = new int[2];
        Check(count_same + count_zero, 0);
        Check(count_same, 1);
        return answer;
    }
}
