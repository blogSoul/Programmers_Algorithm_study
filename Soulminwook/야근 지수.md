import java.util.*;

class Solution {
    public long solution(int n, int[] works) {
        int sum = 0;
        long answer = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
        for(int work : works){
            pq.offer(work);
            sum += work;
        }
        if(sum < n) return 0;
        for(int i = 0; i < n; i++){
            pq.offer(pq.poll() - 1);
        }
        for(int work : pq){
            answer += work * work;
        }
        return answer;
    }
}
