class Solution {
    public String solution(String s) {
        String answer = "";
        int max = Integer.MIN_VALUE, min = Integer.MAX_VALUE;
        for(String word : s.split(" ")){
            max = Math.max(max, Integer.parseInt(word));
            min = Math.min(min, Integer.parseInt(word));
        }
        answer += min + " " + max;
        return answer;
    }
}
