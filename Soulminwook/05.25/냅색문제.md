        for(int i = 0; i < N; i++){
            list.add(Integer.parseInt(input[i]));
        }
        Left = new ArrayList<>();
        Right = new ArrayList<>();
        dfsLeft(0, 0);
        dfsRight(N / 2, 0);

                    start = mid + 1;
                } else {
                    end = mid;
                }
            }
            answer += start;
        }
        System.out.println(answer);
        br.close();
    }
