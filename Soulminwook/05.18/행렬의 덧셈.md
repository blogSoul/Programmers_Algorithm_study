class Solution {
    public int[][] solution(int[][] arr1, int[][] arr2) {
        int X = arr1.length, Y = arr1[0].length;
        int[][] answer = new int[X][Y];
        for(int i = 0; i < X; i++){
            for(int j = 0; j < Y; j++){
                answer[i][j] = arr1[i][j] + arr2[i][j];
            }    
        }
        return answer;
    }
}
