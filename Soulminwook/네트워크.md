import java.util.*;

class Solution {
    public int[] parent, rank;
    public int find(int x){
        if(parent[x] == x){
            return x;
        }
        return parent[x] = find(parent[x]);
    }
    public void union(int x, int y){
        x = find(x);
        y = find(y);
        if(rank[x] < rank[y]){
            parent[x] = y;
        } else {
            parent[y] = x;
            if(rank[x] == rank[y]){
                rank[x]++;
            }
        }

    }
    public int solution(int n, int[][] computers) {
        int answer = 0;
        parent = new int[n];
        rank = new int[n];
        for(int i = 0; i < n; i++){
            parent[i] = i;
        }
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++){
                if(computers[i][j] == 1 && i != j){
                    union(i, j);
                }
            }
        }
        HashSet<Integer> set = new HashSet<>();
        for(int i = 0; i < n; i++){
            set.add(parent[i]);
        }
        return set.size();
    }
}
